    //비교 연산자
    #include <iostream>
     
    using namespace std;
     
    int main(){
        int a;
        int b;
        cin >> a;
        cin >> b;
        cout << " a = " << a << " b = " <<  b << endl;
        
        // a 는 b보다 크다
        bool p = a > b;
        // a 는 b보다 작다
        bool q = a < b;
        // a 와 b는 같다.
        bool r = (a == b);
        // a 와 b는 다르다
        bool z = (a != b);
     
        cout << " p = " << p << endl;
        cout << " q = " << p << endl;
        cout << " r = " << p << endl;
        cout << " z = " << p << endl;
    }
