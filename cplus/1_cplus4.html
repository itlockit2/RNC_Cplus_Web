    #include <iostream>
     
    using namespace std;
     
    int main() {
        // break: 반복문 한개를 종료
        // 숫자를 입력 한것을 그대로 출력한다
        // 만약 유저가 -1을 입력하면 프로그램 종료
        for (int i = 1; true; i++) {
            int num;
            cin >> num;
            cout << i <<" 번째 숫자 " << num << endl;
            if (num == -1) {
                break;
            }
        }
    }
